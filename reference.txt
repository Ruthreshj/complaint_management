User
- id: Long
- name: String
- email: String
- password: String
- role: ENUM (USER, EMPLOYEE, MANAGER)
- createdAt: LocalDateTime

Complaint
- id: Long
- title: String
- description: String
- status: ENUM
- priority: ENUM
- createdBy: User
- assignedTo: Employee (nullable)
- assignedBy: Manager (nullable)
- createdAt: LocalDateTime
- updatedAt: LocalDateTime

ResolutionLog
- id: Long
- complaint: Complaint
- updatedBy: Employee
- comment: String
- status: ENUM
- timestamp: LocalDateTime
